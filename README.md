   var draw = _.draw({
        canvasid: "myCanvas",
        fullscreen: true
    });
    var opt={
        shape:{
            shape:"polygon",
            num:3,
            r:10,
            fill:true,
            a:0
        },
        group:{
            group:"polygon",
            num:7,
            r:30,
            colorful:"circle",
            rotation:true
        },
        motion:{
            motion:"rotate",
            speed:0.1
        }
    }
    draw.setup(opt)
